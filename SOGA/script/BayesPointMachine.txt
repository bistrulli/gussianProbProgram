data feat1 = [63, 16, 28, 55, 22, 20];
data feat2 = [38, 23, 40, 27, 18, 40];
data feat3 = [1, 1, 1, 1, 1, 1];

array[3] w;
array[6] mean;
array[6] res;

for i < 3 {
    w[i] = gm([1.], [0.], [1.]);
} end for;

for i < 6 {
    mean[i] = feat1[i]*w[0] + feat2[i]*w[1] + feat3[i]*w[2];
    res[i] = gm([1.], [0.], [0.1]) + mean[i];
} end for;

observe(res[0] > 0);
observe(res[1] < 0);
observe(res[2] > 0);
observe(res[3] > 0);
observe(res[4] < 0);
observe(res[5] < 0);
