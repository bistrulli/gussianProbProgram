/* assign sources */
s1s1 = gm([0.5,0.5], [1,0], [0,0]);
s2s1 = gm([0.9,0.1], [1,0], [0,0]);

/* enforces constraint about different sources */
observe(s1s1 + s2s1 < 2);
skip;
  
/* assigns distributions of distances */
if s1s1 == 1
{
	d11 = gm([1],[0],[1]);
    d12 = gm([1],[1],[1]);
}
else
{
	d11 = gm([1],[1],[1]);
    d12 = gm([1],[0],[1]);
}
end if;
merge;
skip;

if s2s1 == 1
{
	d21 = gm([1],[0],[1]);
    d22 = gm([1],[1],[1]);
}
else
{
	d21 = gm([1],[1],[1]);
	d22 = gm([1],[0],[1]);
}
end if;
merge;
skip;

/* depending on distances constraints to observations */
observe(d11 > 0.25);
observe(d11 < 0.75);

observe(d12 > 1.25);
observe(d12 < 1.75);

observe(d21 > 1.25);
observe(d21 < 1.75);

observe(d22 > 0.25);
observe(d22 < 0.75);

skip;
