alice = gm([0.33,0.66],[1.,0.5],[1.,0.5]);
bob = gm([0.33,0.66],[1.,0.5],[1.,0.5]);

if alice < 0 {
    absalice = 0 - alice;
} else {
    absalice = alice;
}end if;
merge;

if bob < 0 {
    absbob = 0 - bob;
} else {
    absbob = bob;
} end if;
merge;

if absalice - absbob > 0 {
    obs = 0.6;
} else {
    obs = 0.4;
} end if;
merge;

joint = alice*obs;
